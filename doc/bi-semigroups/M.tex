\section{Definition}

Let $\leq$ be a pre-order on some set $S$ with at least two elements. 
For $X \subseteq S$, define 
\[
\min_\leq(X) \equiv \{x \in X \mid \forall y \in X,\  \neg(y < x)\}. 
\] 
Define 
\[
{\cal P}_{\mathrm{fin}}(S,\ \lesssim) 
   \equiv 
\{X \subseteq S\mid \mbox{$X$ finite and }\min_\leq(X) = X\} 
\] 
and 
\[
  A\ \lift B  \equiv \{a\otimes b\mid a\in A,\ b\in B\}
\] 
and
\[
  A\ \minlift B  	\equiv \min_\leq(A \lift B)
\] 
and
\[
  X\ \minunion Y  \equiv  \min_\leq(X \cup Y) 
\]
then
\[
\mathcal{M}(S,\ \leq,\ \otimes) \equiv ({\cal P}_{\mathrm{fin}}(S,\ \leq),\ \minlift,\ \minunion)
\]

\subsection{Left absorptive}

$??? \Rightarrow (\propname{LAB}(\mathcal{M}(S,\ \otimes,\ \leq)) \Leftrightarrow ???)$

\proof

Consider arbitrary $X, Y$
\begin{align*}
X \minunion (X \minlift Y) 	& = \min_\leq(X \cup \min_\leq(X \lift Y)) \\
							& = \min_\leq(X \cup (X \lift Y)) \\
							& = \min_\leq(X) \\
							& = X
\end{align*}
The second equality holds by the commutativity of $\cup$.



\subsection{Right absorptive}

$??? \Rightarrow (\propname{RAB}(\mathcal{N}(S,\ \otimes,\ \leq)) \Leftrightarrow ???)$

\proof

Consider arbitrary $X, Y$
\begin{align*}
X \minlift (Y \minunion X) 	& = \min_\leq(X \lift \min_\leq(Y \cup X)) \\
							& = \min_\leq(X \lift (Y \cup X)) \\
							& = \min_\leq(X \lift Y \cup X \lift X) \\
							& = \min_\leq(X \lift Y \cup X) \\
							& = \min_\leq(X) \\
							& = X \\
\end{align*}
The second equality holds by the commutativity of $\cup$.
