Theorem correct_to_dual : ∀ (S : Type) (p : A_to S),
      to_dual S (A2C_to S p)
      =
      A2C_to S (A_to_dual S p).
Proof. intros S p.
       unfold to_dual, A_to_dual, A2C_to; simpl.
       rewrite correct_certs_to_dual.
       reflexivity.
Qed.
