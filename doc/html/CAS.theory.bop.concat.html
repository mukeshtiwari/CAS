Lemma bop_concat_not_idempotent :
     ∀ (S : Type) (r : brel S),
        brel_witness S r →
           bop_not_idempotent (list S) (brel_list S r) (bop_concat S).
Proof. intros S r [s _]. unfold bop_not_idempotent. ∃ (s :: nil). simpl.
       rewrite andb_comm. simpl. reflexivity.
Defined.
